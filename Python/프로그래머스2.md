### 문자열 내 p와 y의 개수
```py
def solution(s):
    s = s.lower()
    p_num = s.count("p")
    y_num = s.count("y")
    if p_num == y_num:
        return True
    else:
        return False

print(solution("pPoooyY")) # true
print(solution("Pyy")) # false
```

### 이상한 문자 만들기
```py
def solution(s):
    words = s.split(" ") # ["try", "hello", "world"]
    lst = [] 
    for word in words:
        answer = ''
        for i in range(len(word)):
            if i % 2 == 0: # 짝수번째 알ㅍㅏ벳은
                answer += word[i].upper() # 대문자로 
            else: # 홀수번째 알파벳은 
                answer += word[i].lower() # 소문자로 
        lst.append(answer)
    return " ".join(lst)

print(solution("try hello world")) # "TrY HeLlO WoRlD"
```

### 최대공약수와 최소공배수
```py
def solution(n, m):
    answer = []
    for i in range(min(n, m), 0, -1):
        if n % i == 0 and m % i == 0:
            answer.append(i)
            break
    for i in range(max(n, m), (n*m)+1):
        if i % n == 0 and i % m == 0:
            answer.append(i)
            break
    return answer
```

### 모의고사
```py
student1 = [1,2,3,4,5]
student2 = [2,1,2,3,2,4,2,5]
student3 = [3,3,1,1,2,2,4,4,5,5]
def solution(answers): 
    answer = []
    score1 = 0
    score2 = 0
    score3 = 0
    for i, j in enumerate(answers):
        if student1[i % len(student1)] == j:
            score1 += 1
        if student2[i % len(student2)] == j:
            score2 += 1
        if student3[i % len(student3)] == j:
            score3 += 1
    max_score = max(score1, score2, score3)
    if score1 == max_score:
        answer.append(1)
    if score2 == max_score:
        answer.append(2)
    if score3 == max_score:
        answer.append(3)
    return answer

print(solution([1,2,3,4,5])) # [1]
print(solution([1,3,2,4,2])) # [1,2,3]
```
